```javascript
// 1 添加一个或多个类名
element.classList.add('classone','classtwo',···);


// 2 删除一个或多个类名
element.classList.remove('classone','classtwo',···);


// 3 检索该类里是否有哪个类名
element.classList.indexOf('classname');
element.classList.contains('classname');  // return true or false
```
